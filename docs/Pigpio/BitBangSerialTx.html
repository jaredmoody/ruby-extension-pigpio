def write(buf)
  IF.wave_clear(@tx.pi)
  IF.wave_add_new(@tx.pi)
  ret=IF.wave_add_serial(@tx.pi,@tx.gpio,@baud,@data_bits,@stop_half_bits,0,buf)
  return ret if ret < 0
  wid=IF.wave_create(@tx.pi)
  return wid if wid < 0
  IF.wave_send_once(@tx.pi,wid)
end

def write_sync(buf)
  ret=write(buf)
  return ret if ret < 0
  sync
  IF.wave_clear(@tx.pi)
end
